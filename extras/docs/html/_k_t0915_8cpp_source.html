59 uint16_t KT0915::getRegister(uint8_t reg) {
60 
61  word16_to_bytes result;
62 
63  Wire.beginTransmission(this->deviceAddress);
64  Wire.write(reg);
65  Wire.endTransmission();
66 
67  delayMicroseconds(3000);
68  Wire.requestFrom(this->deviceAddress, 2);

161  * @ingroup GA03
162  * @todo You need to check mechanical Tune features
163  * @brief Sets Tune Dial Mode Interface On
164  * @details This method sets the KT0915 to deal with a mechanical tuning via an external 100K resistor.
165  * @details KT0915 supports a unique Dial Mode (mechanical tuning wheel with a variable resistor) which is
166  * @details enabled by GPIO1 to 2 (10). The dial can be a variable resistor with the tap connected to CH (pin 1).
167  *
168  * @see KT0915; Monolithic Digital FM/MW/SW/LW Receiver Radio on a Chip(TM); pages 19 and 20.
169  *
170  * @param minimu_frequency Start frequency for the user band
171  * @param maximum_frequency Final frequency for the user band
172  */
173 void KT0915::setTuneDialModeOn(uint32_t minimu_frequency, uint32_t maximum_frequency)
174 {
175  kt09xx_amsyscfg reg;
176  kt09xx_gpiocfg gpio;
177  kt09xx_userstartch uf;
178  kt09xx_userguard ug;
179  kt09xx_userchannum uc;
180 
181  reg.raw = getRegister(REG_AMSYSCFG); // Gets the current value from the register
182  reg.refined.USERBAND = this->currentDialMode = DIAL_MODE_ON;
183  setRegister(REG_AMSYSCFG, reg.raw); // Strores the new value in the register
184 
185  gpio.raw = getRegister(REG_GPIOCFG); // Gets the current value from the register
186  gpio.refined.GPIO1 = 2; // Sets Dial Mode interface (pin 1/CH)
187  setRegister(REG_GPIOCFG, gpio.raw); // Stores the new value in the register
188 
189  // TODO: Sets the frequency limits for user and
190 
191  if ( this->currentMode == MODE_AM) {
192  uf.refined.USER_START_CHAN = minimu_frequency;
193  uc.refined.USER_CHAN_NUM = (maximum_frequency - minimu_frequency) / this->currentStep;
194  ug.refined.USER_GUARD = 0x0011;
195  } else {
196  uf.refined.USER_START_CHAN = minimu_frequency / 50;
197  uc.refined.USER_CHAN_NUM = ((maximum_frequency - minimu_frequency) / 50) / this->currentStep;
198  ug.refined.USER_GUARD = 0x001D;
199  }
200 
201  setRegister(REG_USERSTARTCH, uf.raw);
202  setRegister(REG_USERGUARD, ug.raw);
203  setRegister(REG_USERCHANNUM, uc.raw);
204 
205 };

274  * // Set RESET_PIN to -1 if you are using the Arduino RST pin; Select OSCILLATOR_32KHZ, OSCILLATOR_12MHZ etc
275  * // radio.setup(RESET_PIN); Instead the line below, if you use this line, the crystal type considered will be 32.768KHz.
276  * radio.setup(RESET_PIN, OSCILLATOR_12MHZ, REF_CLOCK_DISABLE );

278  * @endcode

280  * Crystal type table
281  * | Dec | binary | Description | defined constant |
282  * | -- | ------ | ----------- | --------------- |
283  * | 0 | 0000 | 32.768KHz | OSCILLATOR_32KHZ |
284  * | 1 | 0001 | 6.5MHz | OSCILLATOR_6_5MHZ |

293  *
